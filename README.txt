━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
  ВАРИАНТ 1: ЛОКАЛЬНАЯ СЕТЬ (Wi-Fi)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

1. Один компьютер запускает сервер:
   запуск_сервер.bat  (или: python server.py)

2. Узнай IP своего компьютера:
   CMD → ipconfig → найди "IPv4 Address" (напр. 192.168.1.10)

3. Остальные игроки — меняют SERVER_URL в bunker_client.py:
   SERVER_URL = "ws://192.168.1.10:8765"

4. Все запускают клиент:
   запуск_игра.bat


━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
  ВАРИАНТ 2: RAILWAY (ИНТЕРНЕТ, БЕСПЛАТНО)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

Деплой сервера:

1. Зарегистрируйся на https://railway.app (бесплатно)

2. Создай новый проект → "Deploy from GitHub"
   (залей папку bunker/ на GitHub сначала)
   
   ИЛИ: используй Railway CLI:
   npm install -g @railway/cli
   railway login
   cd bunker/
   railway init
   railway up

3. В Railway → Variables → добавь:
   PORT = 8765

4. Railway даст тебе URL вида:
   bunker-production-xxxx.up.railway.app

5. Измени SERVER_URL в bunker_client.py:
   SERVER_URL = "wss://bunker-production-xxxx.up.railway.app"
   (wss:// — с SSL для Railway!)

6. Раздай bunker_client.py + data/ всем игрокам

  УСТАНОВКА ЗАВИСИМОСТЕЙ
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

Клиент (каждый игрок):
  pip install PyQt6 websockets

Сервер:
  pip install websockets

  КАК ИГРАТЬ
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

ВЕДУЩИЙ:
  1. Запускает клиент → "СОЗДАТЬ ЛОББИ"
  2. Вводит своё имя + количество мест в бункере
  3. Получает 3-значный код (например: A7X)
  4. Диктует код всем игрокам
  5. Ждёт пока все войдут → нажимает "НАЧАТЬ ИГРУ"
  6. Управляет фазами: выступления → голосование

ИГРОКИ:
  1. Запускают клиент → "ВОЙТИ ПО КОДУ"
  2. Вводят имя + код от ведущего
  3. Видят свою карточку (только свою!)
  4. Нажимают "👁 Раскрыть" на нужном поле публично
  5. При голосовании — решает ведущий

  ❤️ Здоровье    — из health.txt
